-- Create the job_listings table using the defined enums
CREATE TABLE vacancies (
    id bigint primary key generated always as identity,
    title text NOT NULL,
    company text NOT NULL,
    location location_enum NOT NULL,
    salary numeric NOT NULL,
    currency currency_enum NOT NULL,
    job_type job_type_enum NOT NULL,
    experience_level experience_level_enum NOT NULL,
    posted_date timestamp with time zone NOT NULL,
    deadline timestamp with time zone NOT NULL,
    remote boolean NOT NULL,
    skills_required skills_enum[] NOT NULL,
    education_required text NOT NULL,
    benefits benefits_enum[] NOT NULL,
    contact_email text NOT NULL,
    company_website text NOT NULL,
    number_of_vacancies integer NOT NULL
) WITH (OIDS=FALSE);

-- Enable Row Level Security
ALTER TABLE vacancies ENABLE ROW LEVEL SECURITY;
